************ make desktop icon **************

file location : ~/.local/share/applications/<name>.desktop
file content :
	[Desktop Entry]
	Type=Application
	Terminal=false
	Name=Android Studio
	Icon=/opt/android-studio/bin/studio.svg
	Exec=/opt/android-studio/bin/studio.sh



************ how to install ***********
0. ubuntu manual installation folder : /opt or /usr/local

0. change permission
  sudo chmod -R 755 ./
  chmod +x --> chmod ugo+x
  chmod 755 --> chmod u=rwx,go=rx
  . ~/.profile

1. *.deb file
  a. software center 
  b. sudo apt install ./<file>.deb
  c. sudo dpkg -i <file>.deb
     sudo apt-get install -f

2. *.tar.gz or *.tar.xz
  a. unzip
  b. move files to opt folder
  c. set env path
  d. make icon file
